functions can even return the values
write a function that accepts a string and returns no.of characters

function characters (s)
{
    if (typeof (s) == 'string')
    {
        return s.length
    } 
    else
    {
        return "input only strings"
    }
}
let value = characters("prasad")
console.log(value)
let value2 = characters(1234);
console.log(value2);

callback functions:
       callback functions are the functions which are passed as a parameter to another function
       function mad(){
        console.log("i'm mad");
       }
       function mainFun(x){
             x();
       }
       mainFun(mad);



function greet (name, fun)
{
    fun(name); 
}
function sayName (n)
{
    console.log(`My name is ${n}`)
}
greet("prasad", sayName);

sayName is a function which is passed in the parameter of greet , so sayName is a callback function and greet (function which receives callback) is Higher order function

Closures :Closure is a memory technique in which inner loops remember the outer loop values even after outer execution ended.

function outer ()
{
    let a = 5;
    function inner ()
    {
        ++a;
        console.log(a);
    }
    return inner;
}
